基础用法

// React 组件示例
import React from 'react';
import { BulletScreen } from 'rc-bullets-ts';
const App = () => {
const [bullets, setBullets] = React.useState([]);
const addBullet = (text) => {
setBullets(prev => [...prev, {
id: Date.now(),
text,
color: '#fff',
fontSize: 16,
speed: 5
}]);
};
return (
<div className="bullet-container">
<BulletScreen
bullets={bullets}
width={800}
height={400}
autoPlay={true}
loop={false}
/>
<button onClick={() => addBullet('Hello!')}>
发送弹幕
</button>
</div>
);
};
export default App;
